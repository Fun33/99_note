DROP TABLE AUTONUMBER;
/
CREATE TABLE AUTONUMBER  (
	ROW_ID NUMBER(38) NOT NULL PRIMARY KEY,
	M_ROWID NUMBER(38),
	A_ROWID NUMBER(38),
	P_ROWID NUMBER(38)

) ;
        CREATE SEQUENCE AUTONUMBER_SEQ
                MINVALUE 1
                INCREMENT BY 1
                START WITH 1
                NOCACHE
                NOCYCLE;
/
CREATE TRIGGER BI_AUTONUMBER
BEFORE INSERT ON AUTONUMBER
FOR EACH ROW

BEGIN 

SELECT AUTONUMBER_SEQ.NEXTVAL INTO :NEW.ROW_ID FROM DUAL;

END;
/
